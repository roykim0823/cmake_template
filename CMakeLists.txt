cmake_minimum_required(VERSION 3.29)

# C++ modules support is broken with clang-tidy as of CMake 3.29; revisit later.
cmake_policy(SET CMP0155 OLD)

# Default to C++20 unless a parent project has already chosen one.
if(NOT DEFINED CMAKE_CXX_STANDARD)
  set(CMAKE_CXX_STANDARD 20)
endif()
# -std=c++20 (not -std=gnu++20) so -Wpedantic stays clean with PCH.
set(CMAKE_CXX_EXTENSIONS OFF)

project(
  myproject
  VERSION 0.0.2
  DESCRIPTION ""
  LANGUAGES CXX C)

include(cmake/PreventInSourceBuilds.cmake)
include(cmake/ProjectOptions.cmake)
setup_project()

# Hide library symbols by default — opt in to export with the
# SAMPLE_LIBRARY_EXPORT macro generated by generate_export_header.
set(CMAKE_CXX_VISIBILITY_PRESET hidden)

set(GIT_SHA "Unknown" CACHE STRING "SHA this build was generated from")
string(SUBSTRING "${GIT_SHA}" 0 8 GIT_SHORT_SHA)

target_compile_features(options INTERFACE cxx_std_${CMAKE_CXX_STANDARD})

add_library(${PROJECT_NAME}::options ALIAS options)
add_library(${PROJECT_NAME}::warnings ALIAS warnings)

add_subdirectory(configured_files)
add_subdirectory(src)

# Tests, fuzz tests, and packaging only run when this is the top-level project.
if(NOT PROJECT_IS_TOP_LEVEL)
  return()
endif()

include(CTest)
if(BUILD_TESTING)
  add_subdirectory(test)
endif()

if(BUILD_FUZZ_TESTS)
  if(NOT ENABLE_SANITIZER_ADDRESS AND NOT ENABLE_SANITIZER_THREAD)
    message(WARNING "Fuzz tests are most useful with asan or tsan enabled.")
  endif()
  add_subdirectory(fuzz_test)
endif()

if(CMAKE_SKIP_INSTALL_RULES)
  return()
endif()

include(cmake/PackageProject.cmake)
package_project(
  TARGETS
  app
  options
  warnings)

# Embed compiler/version/SHA in the package name to make ABI bugs traceable.
set(CPACK_PACKAGE_FILE_NAME
    "${CMAKE_PROJECT_NAME}-${CMAKE_PROJECT_VERSION}-${GIT_SHORT_SHA}-${CMAKE_SYSTEM_NAME}-${CMAKE_BUILD_TYPE}-${CMAKE_CXX_COMPILER_ID}-${CMAKE_CXX_COMPILER_VERSION}")
include(CPack)
